\chapter[Reproducing time-course analyses]{Reproducing time-course analyses}


\section{Introduction}\label{S:Intro}

This document includes the supplemental data referred to in \ref{chapter:imp}
as well as instructions for how to reproduce our analyses and explore the
data further.
This is a functional document in that scripts (written in the
R programming languange with the \texttt{knitr} package) are embedded, and they were run during
PDF creation to produce the figures and output shown. Thus, the analyses can
be repeated by downloading the source of this document (with the data) or copy-pasting
all of the code into one's own R console.

\section{ Reproducing Figures }\label{S:ReFig}

Below are the scripts to reproduce the figures. The data files
and annotation file needed for the scripts to work are included in
the supplemental data folder. The figures below were exported to PDFs.
Cosmetic alterations (colors, line widths, axes labeling, legends, etc.) were
made with Adobe Illustrator.

\subsection{ Figure 2 }

\singlespacing \begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlcom{#### Since this figure includes data from multiple experiments and files,}
\hlcom{#### they must all be processed first}

\hlcom{# Only select wells seeded with at most 6,000 HCT8 cells}
\hlstd{hct8} \hlkwb{=} \hlkwd{retrieveWells}\hlstd{(wells,} \hlkwc{file} \hlstd{=} \hlstr{"HCT8.txt"}\hlstd{,} \hlkwc{compounds} \hlstd{=} \hlstr{"HCT8"}\hlstd{,} \hlkwc{max.concentrations} \hlstd{=} \hlnum{6000}\hlstd{)}

\hlcom{# The rest of the wells come from these files}
\hlstd{files} \hlkwb{=} \hlkwd{list}\hlstd{(} \hlstr{"CHO.txt"}\hlstd{,} \hlstr{"IMCE.txt"}\hlstd{,} \hlkwd{c}\hlstd{(}\hlstr{"HUVEC-a.txt"}\hlstd{,}\hlstr{"HUVEC-b.txt"}\hlstd{),}
              \hlkwd{c}\hlstd{(}\hlstr{"T84-a.txt"}\hlstd{,}\hlstr{"T84-b.txt"}\hlstd{))}
\hlstd{subsets} \hlkwb{=} \hlkwd{lapply}\hlstd{(files,} \hlkwa{function}\hlstd{(}\hlkwc{f}\hlstd{)} \hlkwd{retrieveWells}\hlstd{(wells,} \hlkwc{file} \hlstd{= f))}
\hlstd{subsets} \hlkwb{=} \hlkwd{c}\hlstd{(}\hlkwd{list}\hlstd{(hct8), subsets)}

\hlcom{# Normalize each subset}
\hlstd{xlims} \hlkwb{=} \hlkwd{list}\hlstd{(}\hlkwd{c}\hlstd{(}\hlopt{-}\hlnum{0.1}\hlstd{,} \hlnum{43}\hlstd{),} \hlkwd{c}\hlstd{(}\hlopt{-}\hlnum{1}\hlstd{,} \hlnum{Inf}\hlstd{),} \hlkwd{c}\hlstd{(}\hlopt{-}\hlnum{1}\hlstd{,} \hlnum{Inf}\hlstd{),} \hlkwd{c}\hlstd{(}\hlopt{-}\hlnum{1}\hlstd{,} \hlnum{60}\hlstd{),} \hlkwd{c}\hlstd{(}\hlopt{-}\hlnum{1}\hlstd{,} \hlnum{27}\hlstd{))}
\hlstd{n.subsets} \hlkwb{=} \hlkwd{mapply}\hlstd{(normalize_toxin, subsets,} \hlkwc{xlim} \hlstd{= xlims)}

\hlcom{# Add a smoother to each subset}
\hlstd{x.scales} \hlkwb{=} \hlkwd{c}\hlstd{(} \hlnum{2}\hlopt{/}\hlnum{3}\hlstd{,} \hlnum{1}\hlstd{,} \hlnum{1}\hlstd{,} \hlnum{2}\hlopt{/}\hlnum{3}\hlstd{,} \hlnum{2}\hlopt{/}\hlnum{3} \hlstd{)}
\hlstd{s.subsets} \hlkwb{=} \hlkwd{mapply}\hlstd{( smoother_toxin, n.subsets,} \hlkwc{x.scale}\hlstd{=x.scales )}

\hlcom{# Calculate ABC for each subset (with different integration limits)}
\hlstd{left} \hlkwb{=} \hlkwd{rep}\hlstd{(}\hlnum{0}\hlstd{,} \hlnum{5}\hlstd{)}
\hlstd{right} \hlkwb{=} \hlkwd{c}\hlstd{(}\hlnum{43}\hlstd{,} \hlnum{40}\hlstd{,} \hlnum{80}\hlstd{,} \hlnum{80}\hlstd{,} \hlnum{27}\hlstd{)}
\hlstd{i.subsets} \hlkwb{=} \hlkwd{mapply}\hlstd{(integrate, s.subsets, left, right)}
\hlstd{allwells} \hlkwb{=} \hlkwd{do.call}\hlstd{(c, i.subsets)}

\hlcom{# Calculate MaxS for each well}
\hlstd{allwells} \hlkwb{=} \hlkwd{max.rate}\hlstd{(allwells,} \hlkwc{ID} \hlstd{=} \hlstr{"toxinAdd"}\hlstd{,} \hlkwc{min.diff} \hlstd{=} \hlnum{10}\hlopt{/}\hlnum{60}\hlopt{/}\hlnum{60}\hlstd{,}
                    \hlkwc{ylim} \hlstd{=} \hlnum{0.8}\hlstd{,} \hlkwc{xlim} \hlstd{=} \hlnum{2}\hlstd{)}



\hlcom{######### Panel A. Different cell types. Same concentrations. #########}
\hlstd{subset} \hlkwb{=} \hlkwd{retrieveWells}\hlstd{(allwells,} \hlkwc{compounds} \hlstd{=} \hlstr{"TcdA"}\hlstd{,} \hlkwc{ID} \hlstd{=} \hlstr{"toxinAdd"}\hlstd{,}
                       \hlkwc{max.concentrations} \hlstd{=} \hlnum{101}\hlstd{,} \hlkwc{min.concentrations} \hlstd{=} \hlnum{99}\hlstd{)}
\hlstd{panelA} \hlkwb{=} \hlkwd{plot}\hlstd{(subset,} \hlkwc{xlim} \hlstd{=} \hlkwd{c}\hlstd{(}\hlopt{-}\hlnum{1}\hlstd{,} \hlnum{10}\hlstd{),} \hlkwc{se} \hlstd{=} \hlnum{FALSE}\hlstd{,} \hlkwc{color} \hlstd{=} \hlstr{"by.total.compounds"}\hlstd{,}
              \hlkwc{linetype} \hlstd{=} \hlstr{"by.compounds"}\hlstd{)}

\hlcom{######### Panel B. Same cell type. Different toxins ############}
\hlstd{subset} \hlkwb{=} \hlkwd{retrieveWells}\hlstd{(allwells,} \hlkwc{compounds} \hlstd{=} \hlstr{"IMCE"}\hlstd{)}
\hlstd{subset2} \hlkwb{=} \hlkwd{retrieveWells}\hlstd{(subset,} \hlkwc{compounds} \hlstd{=} \hlkwd{c}\hlstd{(}\hlstr{"TcdA"}\hlstd{,} \hlstr{"TcdB"}\hlstd{),} \hlkwc{ID} \hlstd{=} \hlstr{"toxinAdd"}\hlstd{,}
                        \hlkwc{max.concentrations} \hlstd{=} \hlkwd{c}\hlstd{(}\hlnum{101}\hlstd{,} \hlnum{101}\hlstd{),} \hlkwc{min.concentrations} \hlstd{=} \hlkwd{c}\hlstd{(}\hlnum{99}\hlstd{,} \hlnum{99}\hlstd{))}
\hlstd{panelB} \hlkwb{=} \hlkwd{plot}\hlstd{(subset2,} \hlkwc{xlim} \hlstd{=} \hlkwd{c}\hlstd{(}\hlopt{-}\hlnum{0.1}\hlstd{,} \hlnum{2}\hlstd{),} \hlkwc{se} \hlstd{=} \hlnum{FALSE}\hlstd{)}

\hlcom{######## Panel C. MaxS and ABC for IMCE cells ###########}
\hlstd{subset} \hlkwb{=} \hlkwd{retrieveWells}\hlstd{(allwells,} \hlkwc{compounds} \hlstd{=} \hlstr{"IMCE"}\hlstd{)}
\hlstd{MaxS} \hlkwb{=} \hlkwd{groupMetric}\hlstd{(subset,} \hlkwc{ID} \hlstd{=} \hlstr{"toxinAdd"}\hlstd{,} \hlkwc{metric} \hlstd{=} \hlstr{"max.rate"}\hlstd{)}
\hlstd{p1} \hlkwb{=} \hlkwd{plotMetric}\hlstd{(MaxS)}
\hlstd{ABC} \hlkwb{=} \hlkwd{groupMetric}\hlstd{(subset,} \hlkwc{ID} \hlstd{=} \hlstr{"toxinAdd"}\hlstd{,} \hlkwc{metric} \hlstd{=} \hlstr{"integral"}\hlstd{)}
\hlstd{p2} \hlkwb{=} \hlkwd{plotMetric}\hlstd{(ABC)}
\hlstd{panelC} \hlkwb{=} \hlkwd{arrangeGrob}\hlstd{(p1} \hlopt{+} \hlkwd{theme}\hlstd{(}\hlkwc{legend.position} \hlstd{=} \hlstr{"none"}\hlstd{),}
                     \hlstd{p2} \hlopt{+} \hlkwd{theme}\hlstd{(}\hlkwc{legend.position} \hlstd{=} \hlstr{"none"}\hlstd{),} \hlkwc{ncol} \hlstd{=} \hlnum{2}\hlstd{)}

\hlcom{####### Panel D. The MCC for each cell type ##############}
\hlcom{# The MCC was found by plotting the ABC of each cell type over}
\hlcom{# a range of concentrations. The concentration diverging from}
\hlcom{# controls was considered the MCC}
\hlstd{ABC} \hlkwb{=} \hlkwd{groupMetric}\hlstd{(allwells,} \hlkwc{ID} \hlstd{=} \hlstr{"toxinAdd"}\hlstd{,} \hlkwc{metric} \hlstd{=} \hlstr{"integral"}\hlstd{)}
\hlstd{ABC}\hlopt{$}\hlstd{cells} \hlkwb{=} \hlkwd{groupWells}\hlstd{(allwells,} \hlkwc{group} \hlstd{=} \hlstr{"by.total.compounds"}\hlstd{)}
\hlkwd{plotMetric}\hlstd{(ABC,} \hlkwc{file} \hlstd{=} \hlnum{FALSE}\hlstd{)} \hlopt{+} \hlkwd{facet_wrap}\hlstd{(}\hlopt{~}\hlstd{cells,} \hlkwc{scales} \hlstd{=} \hlstr{"free_y"}\hlstd{)}
\end{alltt}
\end{kframe}
\includegraphics[width=\maxwidth]{apximp/preFigure2} 

\end{knitrout} \doublespacing
